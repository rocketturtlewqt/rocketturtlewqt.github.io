response

  • 若不设置response的头信息，浏览器默认以html形式解析字符串
  • writeHead
  1. 第一个参数为状态码
  2. 第二个参数用来设置头部具体信息，例如，content-type
const http = require('http');

const server = http.createServer((req, res) => {
  res.writeHead(200, {
    'content-type': 'text/plain'
  })
  res.write('<div>hello world</div>')
  res.end();
});

server.listen('7002');
