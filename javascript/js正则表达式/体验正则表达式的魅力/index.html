1 min read 
let str = 'asjdhka17821sdjashdkj23324';
//将数字拼接成字符串返回
//不用正则表达式
function parseNumber(s) {
  let result = '';
  for (let i = 0; i < s.length; i++){
    if (!Number.isNaN(parseInt(s[i]))) result += s[i];
  }
  return result;
}
